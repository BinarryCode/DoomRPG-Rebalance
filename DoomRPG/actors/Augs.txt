// Small Battery
actor BatterySmall : CustomInventory
{
    Tag "Small Battery - Restores 10% of your Augmentation Battery"
    
    +INVBAR
    
    Inventory.MaxAmount 1000
    Inventory.InterHubAmount 1000
    Inventory.PickupMessage "You got a Small Battery."
    Inventory.PickupSound "items/battery"
    Inventory.UseSound "items/battery"
    Inventory.Icon "BATT1"
    
    Scale 0.75
    
    States
    {
    Spawn:
        BATT A -1 Bright
        Stop
	Pickup:
		TNT1 A 0 ACS_NamedExecuteWithResult("CheckCapacity")
		Stop
    Use:
		TNT1 A 0 A_JumpIf(ACS_NamedExecuteWithResult("CheckAugBatteryMax") == 1, "Spawn")
        TNT1 A 0 ACS_NamedExecuteAlways("AddBattery", 0, 10)
        Stop
    }
}
