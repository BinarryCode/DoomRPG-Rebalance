actor ScriptedMarineRPG : ScriptedMarine Replaces ScriptedMarine
{
	Speed 16
	
	Tag "Marine"
	
	States
	{
	Spawn:
		PLAY A 0 NoDelay ACS_NamedExecuteAlways("MonsterInit", 0, MF_NEGATIVEXP)
		Goto Super::Spawn
	See:
		PLAY A 0 ACS_NamedExecuteAlways("MonsterInit", 0, MF_NEGATIVEXP)
		Goto Super::See
	Missile.Pistol:
		PLAY E 0 A_FaceTarget
		PLAY F 6 BRIGHT A_M_FirePistol(1)
		PLAY A 0 A_FaceTarget
		PLAY A 0 A_M_Refire
		PLAY A 0
		Goto See
	SkipAttack:
		PLAY A 0
		Goto See
	}
}
