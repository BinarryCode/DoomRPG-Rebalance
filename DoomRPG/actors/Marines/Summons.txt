ACTOR DRPGMarineGuardPistolSummoned
{
  Health 100
  GibHealth -50
  Radius 20
  Height 56
  Speed 8
  PainChance 256
  DamageFactor "Crush", 10
  Monster
  +FLOORCLIP
  +AVOIDMELEE
  +FRIENDLY
  +MISSILEMORE
  -COUNTKILL
  +AMBUSH
  +NEVERRESPAWN
  +QUICKTORETALIATE
  +NOBLOCKMONST
  +NOINFIGHTING
  +DONTHARMSPECIES
  +FRIENDLY
  -SOLID
  +ALWAYSFAST
  +QUICKTORETALIATE
  +MISSILEMORE
  Obituary "%o stood in the line of fire for some reason."
  Species "Player"
  Tag "Marine"
  SeeSound ""
  PainSound "drpgmarinemale2/pain"
  DeathSound "drpgmarinemale2/death"
  ActiveSound "drpgmarinemale/active"
  AttackSound ""
  States
  {
  Spawn:
    TNT1 A 0 NoDelay ACS_NamedExecuteAlways("MonsterInit", 0, MF_NOPAYKILL | MF_NODROPS | MF_NAMEGEN)
	Goto SpawnSearch
  SpawnSearch:
    NMA2 A 3 A_Wander
    TNT1 A 0 A_Look
    NMA2 A 3 A_Wander
    TNT1 A 0 A_Look
    NMA2 B 3 A_Wander
    TNT1 A 0 A_Look
    NMA2 B 3 A_Wander
    TNT1 A 0 A_Look
    NMA2 C 3 A_Wander
    TNT1 A 0 A_Look
    NMA2 C 3 A_Wander
    TNT1 A 0 A_Look
    NMA2 D 3 A_Wander
    TNT1 A 0 A_Look
    NMA2 D 3 A_Wander
    TNT1 A 0 A_Look
    Loop
  See:
    TNT1 A 0 A_JumpIfInventory ("DRPGMarineChaingunShotCounter", 40, "RetreatReload")
    TNT1 A 0 A_ChangeFlag ("FRIGHTENED", 0)
    TNT1 A 0 A_JumpIfHealthLower (30, "SeeScared")
    NMA2 AABBCCDD 3 A_Chase
    Loop
  SeeScared:
    TNT1 A 0 A_JumpIfHealthLower (30, 1)
	Goto See
    TNT1 A 0 A_ChangeFlag ("FRIGHTENED", 1)
    NMA2 AABBCCDD 2 A_Chase
    TNT1 A 0 A_JumpIfCloser (1024, 1)
	Goto Overwatch
    NMA2 A 2 A_Chase
    TNT1 A 0 A_JumpIfCloser (1024, 1)
	Goto Overwatch
    NMA2 B 2 A_Chase
    TNT1 A 0 A_JumpIfCloser (1024, 1)
	Goto Overwatch
    NMA2 B 2 A_Chase
    TNT1 A 0 A_JumpIfCloser (1024, 1)
	Goto Overwatch
    NMA2 C 2 A_Chase
    TNT1 A 0 A_JumpIfCloser (1024, 1)
	Goto Overwatch
    NMA2 C 2 A_Chase
    TNT1 A 0 A_JumpIfCloser (1024, 1)
	Goto Overwatch
    NMA2 D 2 A_Chase
    TNT1 A 0 A_JumpIfCloser (1024, 1)
	Goto Overwatch
    NMA2 D 2 A_Chase
    TNT1 A 0 A_JumpIfCloser (1024, "SeeScared")
	Goto Overwatch
  RetreatReload:
    TNT1 A 0 A_ChangeFlag ("FRIGHTENED", 1)
    NMA2 AA 3 A_Chase ("", "")
    TNT1 A 0 A_JumpIfCloser (180, 2)
    TNT1 A 0 A_Jump (24, "Reload")
    NMA2 BB 3 A_Chase ("", "")
    TNT1 A 0 A_JumpIfCloser (180, 2)
    TNT1 A 0 A_Jump (24, "Reload")
    NMA2 CC 3 A_Chase ("", "")
    TNT1 A 0 A_JumpIfCloser (180, 2)
    TNT1 A 0 A_Jump (24, "Reload")
    NMA2 DD 3 A_Chase ("", "")
    TNT1 A 0 A_JumpIfCloser (180, 2)
    TNT1 A 0 A_Jump (24, "Reload")
    TNT1 A 0
    Goto RetreatReload+1
  Missile:
    TNT1 A 0 A_JumpIfInventory ("DRPGMarinePistolShotCounter", 6, "RetreatReload")
    NMA2 EEEEEE 1 A_FaceTarget (10)
    TNT1 A 0 A_FaceTarget
    TNT1 A 0 A_GiveInventory ("DRPGMarinePistolShotCounter", 1)
    TNT1 A 0 A_PlaySound("drpgmarines/pistol", CHAN_WEAPON, 1, 0, 0.8)
    NMA2 F 4 Bright A_CustomBulletAttack (22.5, 0, 1, random(1,5) * 3, "DRPGMarineBulletPuff", 0, CBAF_NORANDOM)
    NMA2 E 6
    TNT1 A 0 A_JumpIfInventory ("DRPGMarinePistolShotCounter", 6, "RetreatReload")
    NMA2 E 5 A_MonsterRefire (0, "Overwatch")
    Goto Missile
  Reload:
    NMA2 EE 4
    TNT1 A 0 A_PlaySound("drpgmarines/pistolreloadout", 7)
    NMA2 EE 4
    TNT1 A 0 A_ChangeFlag ("FRIGHTENED", 0)
    TNT1 A 0 A_PlaySound("drpgmarines/pistolreloadin", 7)
    TNT1 A 0 A_TakeInventory ("DRPGMarinePistolShotCounter", 6)
    NMA2 EEEE 4
    NMA2 E 3
    TNT1 A 0 A_PlaySound("drpgmarines/pistolcock", 7)
    NMA2 E 6
    Goto See+2
  Overwatch:
    TNT1 A 0 A_JumpIfInventory ("DRPGMarinePistolShotCounter", 6, "RetreatReload")
    TNT1 A 0 A_Chase ("", "Missile", CHF_DONTMOVE)
    NMA2 E 4 A_FaceTarget
    TNT1 A 0 A_Chase ("", "Missile", CHF_DONTMOVE)
    NMA2 E 4 A_FaceTarget
    TNT1 A 0 A_Jump (80, 1)
    Loop
    TNT1 A 0
    Goto See+2
  Pain:
    NMA2 G 3
    NMA2 G 3 A_Pain
    Goto See
  Death:
    NMA2 H 4
    NMA2 I 4 A_Scream
    NMA2 J 4 A_NoBlocking
    NMA2 K 4
    TNT1 A 0 A_SpawnItem ("DRPGBodyThud", 0, 0, 0, 0)
    NMA2 LM 4
    NMA2 N -1
    Stop
  XDeath:
    NMA2 O 4
    NMA2 P 4 A_XScream
    NMA2 Q 4 A_NoBlocking
    TNT1 A 0 A_SpawnItem ("DRPGGibThud", 0, 0, 0, 0)
    NMA2 RSTUV 4
    NMA2 W -1
    Stop
  Crush:
    TNT1 A 1 A_SpawnItem("DRPGCrushedMarine",0,0,0,0)
    Stop
  }
}
