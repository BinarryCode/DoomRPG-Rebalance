// Shield Booster
actor ShieldBooster : CustomInventory
{
	Scale 0.5
	renderStyle Add
	
	-NOGRAVITY
	+FLOATBOB
	-FLOORCLIP
	
	Inventory.PickupMessage "Got a Shield Booster!"
	Inventory.PickupSound "shield/booster"
	
	States
	{
	Spawn:
		SHLD A -1 Bright
		Stop
	Pickup:
		SHLD A 0 ACS_NamedExecute("AddShield", 0, Random(1, 100))
		Stop
	}
}
