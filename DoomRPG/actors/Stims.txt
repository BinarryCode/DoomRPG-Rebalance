actor StimSmall : Inventory
{
	Scale 0.25
	
	Inventory.DefMaxAmount
	Inventory.InterHubAmount 1000
	Inventory.PickupMessage "Got a Small Stim!"
	Inventory.PickupSound "misc/stim"
	Inventory.UseSound "misc/stim"
	
	States
	{
	Spawn:
		STIM B -1 Bright
		Stop
	}
}
