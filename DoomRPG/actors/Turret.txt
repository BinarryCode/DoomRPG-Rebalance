ACTOR DRPGOutpostTurret 19997
{
    Health 150
    Radius 16
    Height 16
    Speed 0
    Painchance 256
    Minmissilechance 1
    Mass 0x7fffffff
    Damagefactor "Melee", 0.2
    Damagefactor "Bullet", 0.2
    Damagefactor "Fire", 0.75
    Damagefactor "Plasma", 1.2
    obituary "%o forgot to check the ceiling."
    painsound ""
    deathsound "drpgturret/death"
    Monster
    +AMBUSH
    +QUICKTORETALIATE
    +SPAWNCEILING
    +CEILINGHUGGER
    +NOGRAVITY
    +NOBLOOD
    +NOSPLASHALERT
    +DONTFALL
    -COUNTKILL
    +NEVERRESPAWN
    +NOINFIGHTING
    +NOTELEOTHER
    +DONTMORPH
    +DONTBLAST
    +SEEINVISIBLE
    Species "Marine"
    Tag "Defense Turret"
    States
    {
    Spawn:
        TNT1 A 0 NoDelay ACS_NamedExecuteAlways("MonsterInit", 0, MF_NOXP | MF_NOSTATS | MF_NOAURA | MF_NODROPS)
        TNT1 A 0
        Goto Idle
    Idle:
        TNT1 A 0 A_CheckFlag("NEVERTARGET", "PoweredDown", AAPTR_DEFAULT)
        TNT1 A 0 A_CheckFlag("NOPAIN", "FuriousIdle", AAPTR_DEFAULT)
        TNT1 A 0 A_PlaySound ("drpgturret/hum", 6, 1, 1, 1.3)
        OTUR A 1
        Loop
    PoweredDown:
        TNT1 A 0 A_ChangeFlag ("NEVERTARGET", 1)
        TNT1 A 0 A_ChangeFlag ("NOPAIN", 0)
        TNT1 A 0 A_StopSound (6)
        OTUR A 1
        Loop
    FuriousIdle:
        TNT1 A 0 A_PlaySound ("drpgturret/hum", 6, 1, 1, 1.3)
        OTUR AAAAAAAAAAAAAAAAAAAA 1 A_Look
        TNT1 A 0 A_Jump (230, "FuriousIdle")
        TNT1 A 0 A_PlaySound ("drpgturret/beep", 7)
        Loop
    See:
        TNT1 A 0 A_CheckFlag("NEVERTARGET", "PoweredDown", AAPTR_DEFAULT)
        TNT1 A 0 A_PlaySound ("drpgturret/hum", 6, 1, 1, 1.3)
        OTUR AAAAAAAAAAAAAAAAAAAA 1 A_Chase ("", "Missile")
        TNT1 A 0 A_Jump (230, "See")
        TNT1 A 0 A_PlaySound ("drpgturret/beep", 7)
        Loop
    Missile:
        TNT1 A 0 A_PlaySound ("drpgturret/targeting", 7)
        OTUR AAAAAAAAAAAAAAA 2 A_FaceTarget (15)
    MissileLoop:
        TNT1 A 0 A_FaceTarget
        TNT1 A 0 A_PlaySound("drpgturret/fire", CHAN_WEAPON, 1, 0, 0.6)
        OTUR B 2 Bright A_CustomBulletAttack (16, 0, 2, random(1,5) * 3, "DRPGMarineBulletPuff", 0, CBAF_NORANDOM)
        OTUR C 2 Bright
        OTUR A 3 A_FaceTarget
        TNT1 A 0 A_MonsterRefire(0, "MissileEnd")
        goto MissileLoop
    MissileEnd:
        TNT1 A 0 A_PlaySound ("drpgturret/cooldown", 5)
        OTUR AAAAA 3 A_FaceTarget
        goto See
    Pain:
        TNT1 A 0 A_CheckFlag("NEVERTARGET", "Stahp", AAPTR_DEFAULT)
        TNT1 A 0 A_CheckFlag("AMBUSH", "GotShot", AAPTR_DEFAULT)
        Goto PainContinue
    PainContinue:
        OTUR A 2 A_Pain
        goto See
    GotShot:
        TNT1 A 0 ACS_NamedExecuteAlways("PissOffMarines", 0)
        Goto Enraged
    Stahp:
        TNT1 A 0 ACS_NamedExecuteAlways("PissOffMarines", 0)
        Goto Enraged
    Enraged:
        TNT1 A 0 A_CheckFlag("NEVERTARGET", "PoweredDown", AAPTR_DEFAULT)
        TNT1 A 0 A_ChangeFlag ("AMBUSH", 0)
        TNT1 A 0 A_ChangeFlag ("NOPAIN", 1)
        TNT1 A 0 A_PlaySound ("drpgturret/activate", 5)
        OTUR A 80
        Goto See
    Death:
        TNT1 A 0 A_CheckFlag("FRIENDLY", "DeathHatred", AAPTR_DEFAULT)
        TNT1 A 0 A_StopSound (6)
        TNT1 A 0 A_Scream
        TNT1 A 0 A_SpawnItemEx ("DRPGOutpostTurretExplosion", 0, 0, -4, 0, 0, 0, 0, 128)
    DeathLoop:
        TNT1 A 0 A_PlaySound ("drpgturret/spark", 5, 1, 0, 1.5)
        TNT1 A 0 A_SpawnItemEx ("DRPGOutpostTurretSpark", frandom(-12, 12), frandom(-12, 12), 2, 0, 0, 0, 0, 128)
        TNT1 A 0 A_Jump (256, 1,2,3,4,5,6,7)
        OTUR DDDDDD 10
        TNT1 A 0
        Goto DeathLoop
    DeathHatred:
        TNT1 A 0 ACS_NamedExecuteAlways("PissOffMarines", 0)
        Goto Death+1
    }
}

ACTOR DRPGPortableTurret
{
    // Upgrades
    var int user_upgrade[128];
    
    // Weapons
    var int user_weapon;
    var int user_bullet_damage;
    var int user_bullet_rof;
    var int user_pellet_damage;
    var int user_pellet_rof;
    var int user_pellet_spread;
    var int user_pellet_amount;
    var int user_rocket_rof;
    var int user_plasma_rof;
    var int user_railgun_damage;
    var int user_railgun_rof;
    var int user_railgun_ripping;
    
    // Ammo
    var int user_ammo_bullet;
    var int user_ammo_shell;
    var int user_ammo_rocket;
    var int user_ammo_plasma;
    var int user_ammo_rail;
    
    // Self Destruct
    var int user_destruct_range;
    var int user_destruct_damage;
    
    // Misc
    var int user_damage_type;
    
    Health 100
    Radius 16
    Height 16
    Speed 0
    Painchance 256
    Mass 0x7FFFFFFF
    Species "Player"
    
    +SOLID
    +NOGRAVITY
    +FLOAT
    +SHOOTABLE
    +NOBLOOD
    +NOCLIP
    +FRIENDLY
    
    States
    {
    Spawn:
        TNT1 A 0 NoDelay ACS_NamedExecuteAlways("TurretInit", 0)
        Goto Idle
    See:
        TNT1 A 0 A_PlaySound("drpgturret/hum", CHAN_6, 1.0, true, 1.25)
        PTUR A 1 A_Chase("", "Missile")
        TNT1 A 0 A_JumpIfTargetInLOS(1)
        Goto Forget
        TNT1 A 0 A_Jump(255, "See")
        TNT1 A 0 A_PlaySound("drpgturret/beep", CHAN_7)
        Loop
    Forget:
        TNT1 A 0 A_ClearTarget
    Idle:
        TNT1 A 0 A_PlaySound("drpgturret/hum", CHAN_6, 1.0, true, 1.25)
        TNT1 A 0 A_Wander
        PTUR A 1 A_Look
        Loop
    Searching:
        PTUR A -1
        Stop
    Missile:
        TNT1 A 0 A_JumpIf(user_weapon == TW_NONE, "See")
        TNT1 A 0 A_JumpIf(user_weapon == TW_BULLET, "Missile.Bullet")
        TNT1 A 0 A_JumpIf(user_weapon == TW_PELLET, "Missile.Pellet")
        TNT1 A 0 A_JumpIf(user_weapon == TW_ROCKET, "Missile.Rocket")
        TNT1 A 0 A_JumpIf(user_weapon == TW_PLASMA, "Missile.Plasma")
        TNT1 A 0 A_JumpIf(user_weapon == TW_RAILGUN, "Missile.Railgun")
        Goto See
    Missile.Bullet:
        TNT1 A 0 A_JumpIf(user_ammo_bullet <= 0, "See")
        TNT1 A 0 A_PlaySound("drpgturret/targeting", CHAN_7)
        PTUR A 2 A_FaceTarget
    Missile.BulletLoop:
        TNT1 A 0 A_FaceTarget
        TNT1 A 0 A_PlaySound("drpgturret/fire", CHAN_WEAPON, 1.0, 0, 0.6)
        PTUR B 0 Bright A_CustomBulletAttack(8, 0, 1, Random(5, 10) * (user_bullet_damage + 1), "BulletPuff", 0, CBAF_NORANDOM)
        TNT1 A 0 ACS_NamedExecuteAlways("TurretUseAmmo", 0, TW_BULLET)
        PTUR B 2 Bright
        PTUR A 0 A_SetTics(6 - user_bullet_rof)
        PTUR A 0 A_FaceTarget
        TNT1 A 0 A_JumpIf(user_ammo_bullet <= 0, "See")
        TNT1 A 0 A_MonsterRefire(0, "MissileEnd")
        Loop
    Missile.Pellet:
        TNT1 A 0 A_JumpIf(user_ammo_shell <= 0, "See")
        TNT1 A 0 A_PlaySound("drpgturret/targeting", CHAN_7)
        PTUR A 2 A_FaceTarget
    Missile.PelletLoop:
        TNT1 A 0 A_FaceTarget
        TNT1 A 0 A_PlaySound("weapons/sshotf", CHAN_WEAPON, 1.0, 0, 0.6)
        PTUR B 0 Bright A_CustomBulletAttack(16 - (user_pellet_spread * 3), 16 - (user_pellet_spread * 3), 3 * (user_pellet_amount + 1), Random(10, 20) * (user_pellet_damage + 1), "BulletPuff", 0, CBAF_NORANDOM)
        TNT1 A 0 ACS_NamedExecuteAlways("TurretUseAmmo", 0, TW_PELLET)
        PTUR B 10 Bright
        PTUR A 0 A_SetTics(35 - (user_pellet_rof * 5))
        PTUR A 0 A_FaceTarget
        TNT1 A 0 A_JumpIf(user_ammo_shell <= 0, "See")
        TNT1 A 0 A_MonsterRefire(0, "MissileEnd")
        Loop
    Missile.Rocket:
        TNT1 A 0 A_JumpIf(user_ammo_rocket <= 0, "See")
        TNT1 A 0 A_PlaySound("drpgturret/targeting", CHAN_7)
        PTUR A 2 A_FaceTarget
    Missile.RocketLoop:
        TNT1 A 0 A_FaceTarget
        PTUR C 0 Bright A_CustomMissile("DRPGTurretRocket", height, 0, 0, CMF_AIMOFFSET | CMF_CHECKTARGETDEAD)
        TNT1 A 0 ACS_NamedExecuteAlways("TurretUseAmmo", 0, TW_ROCKET)
        PTUR C 10 Bright
        PTUR A 0 A_SetTics(70 - (user_rocket_rof * 12))
        PTUR A 0 A_FaceTarget
        TNT1 A 0 A_JumpIf(user_ammo_rocket <= 0, "See")
        TNT1 A 0 A_MonsterRefire(0, "MissileEnd")
        Loop
    Missile.Plasma:
        TNT1 A 0 A_JumpIf(user_ammo_plasma <= 0, "See")
        TNT1 A 0 A_PlaySound("drpgturret/targeting", CHAN_7)
        PTUR A 2 A_FaceTarget
    Missile.PlasmaLoop:
        TNT1 A 0 A_FaceTarget
        PTUR D 0 Bright A_CustomMissile("DRPGTurretPlasma", height, 0, 0, CMF_AIMOFFSET | CMF_CHECKTARGETDEAD)
        TNT1 A 0 ACS_NamedExecuteAlways("TurretUseAmmo", 0, TW_PLASMA)
        PTUR D 3 Bright
        PTUR A 0 A_SetTics(5 - user_plasma_rof)
        PTUR A 0 A_FaceTarget
        TNT1 A 0 A_JumpIf(user_ammo_plasma <= 0, "See")
        TNT1 A 0 A_MonsterRefire(0, "MissileEnd")
        Loop
    Missile.Railgun:
        TNT1 A 0 A_JumpIf(user_ammo_rail <= 0, "See")
        TNT1 A 0 A_PlaySound("drpgturret/targeting", CHAN_7)
        PTUR A 2 A_FaceTarget
    Missile.RailgunLoop:
        TNT1 A 0 A_FaceTarget
        PTUR E 0 Bright A_CustomRailgun(Random(50, 100) * (user_railgun_damage + 1), 0, "LightGreen", "Green", RGF_FULLBRIGHT | user_railgun_ripping, true)
        TNT1 A 0 ACS_NamedExecuteAlways("TurretUseAmmo", 0, TW_RAILGUN)
        PTUR E 10 Bright
        PTUR A 0 A_SetTics(35 - (user_plasma_rof * 5))
        PTUR A 0 A_FaceTarget
        TNT1 A 0 A_JumpIf(user_ammo_rail <= 0, "See")
        TNT1 A 0 A_MonsterRefire(0, "MissileEnd")
        Loop
    MissileEnd:
        TNT1 A 0 A_PlaySound("drpgturret/cooldown", CHAN_5)
        PTUR A 2 A_FaceTarget
        Goto See
    SelfDestruct:
        TNT1 A 0 A_StopSound(CHAN_6)
        TNT1 A 0 A_PlaySound("turret/countdown", CHAN_5, 1.0, false, 0.0)
        PTUR A 350
        TNT1 A 0 A_PlaySound("turret/explode", CHAN_5, 1.0, false, 0.0)
        TNT1 A 0 A_Explode(user_destruct_damage, user_destruct_range)
        TNT1 A 0 ACS_NamedExecuteAlways("TurretPostSelfDestruct", 0)
        Stop
    Pain:
        TNT1 A 0 A_SetUserVar("user_damage_type", DT_NORMAL)
        Goto See
    Pain.Slime:
        TNT1 A 0 A_SetUserVar("user_damage_type", DT_TOXIC)
        Goto See
    Pain.Radiation:
        TNT1 A 0 A_SetUserVar("user_damage_type", DT_RADIATION)
        Goto See
    Pain.Melee:
        TNT1 A 0 A_SetUserVar("user_damage_type", DT_MELEE)
        Goto See
    Pain.Fire:
        TNT1 A 0 A_SetUserVar("user_damage_type", DT_FIRE)
        Goto See
    Pain.Plasma:
        TNT1 A 0 A_SetUserVar("user_damage_type", DT_PLASMA)
        Goto See
    Pain.Lightning:
        TNT1 A 0 A_SetUserVar("user_damage_type", DT_LIGHTNING)
        Goto See
    Death:
        TNT1 A 0 A_SpawnItemEx("TeleportFog")
        Stop
    }
}

actor DRPGTurretRocket : Rocket
{
    Damage (CallACS("TurretGetProjectileDamage", TP_ROCKET))
    Species "Player"
    
    +MTHRUSPECIES
    
    States
    {
	Spawn:
        TNT1 A 0 NoDelay A_JumpIf(CallACS("TurretGetProjectileProperty", TP_ROCKET) > 0, "Seek")
        MISL A 1 Bright
        Loop
    Seek:
        TNT1 A 0 A_ChangeFlag("SEEKERMISSILE", true)
        MISL A 1 Bright A_SeekerMissile(1, 1 * CallACS("TurretGetProjectileProperty", TP_ROCKET), SMF_LOOK)
        Loop
    }
}
