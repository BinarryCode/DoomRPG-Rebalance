actor ScriptedMarineDRPG : ScriptedMarine
{
	Speed 16
	
	-SOLID
	+ALWAYSFAST
	+QUICKTORETALIATE 

	States
	{
	See:
		PLAY ABCD 3 A_MarineChase
		TNT1 A 0 A_GiveInventory("DamageFontGiver")
		Loop
	Missile.Pistol:
		PLAY E 0 A_FaceTarget
		PLAY F 6 BRIGHT A_M_FirePistol(1)
		PLAY A 0 A_FaceTarget
		PLAY A 0 A_M_Refire
		PLAY A 0
		Goto See
	SkipAttack:
		PLAY A 0
		Goto See
	}
}
