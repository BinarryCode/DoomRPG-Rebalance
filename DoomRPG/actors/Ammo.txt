ACTOR DRPGBackpack : CustomInventory
{
    Inventory.PickupMessage "$GOTBACKPACK"
    Inventory.PickupSound "items/pack"
    
    States
    {
    Spawn:
        BPAK A -1
        Stop
    Pickup:
        TNT1 A 0 A_GiveInventory("Clip", 50)
        TNT1 A 0 A_GiveInventory("Shell", 20)
        TNT1 A 0 A_GiveInventory("RocketAmmo", 5)
        TNT1 A 0 A_GiveInventory("Cell", 100)
        Stop
    }
}

actor DRPGBigBackPack : DRPGBackpack
{
    Inventory.PickupMessage "Picked up a Big Golden Backpack loaded with ammo!"
    Inventory.PickupSound "items/pack"
    
    States
    {
    Spawn:
        GPAK A 6 Bright
        GPAK A 1 Bright A_SpawnItemEx("DRPGBigBackpackParticle", Random(-Radius, Radius), Random(-Radius, Radius), Random(-Height + 16, Height + 8))
        Loop
    Pickup:
        TNT1 A 0 A_GiveInventory("Clip", 200)
        TNT1 A 0 A_GiveInventory("Shell", 50)
        TNT1 A 0 A_GiveInventory("RocketAmmo", 50)
        TNT1 A 0 A_GiveInventory("Cell", 300)
        Stop
    }
}

actor DRPGBigBackpackParticle
{
    +NOTIMEFREEZE
    +NOINTERACTION
    +NOTONAUTOMAP
    
    Scale 0.025
    
    RenderStyle Add
    
    States
    {
    Spawn:
        AUEF B 1 Bright
        TNT1 A 0 A_FadeOut(0.05)
        Loop
    }
}
