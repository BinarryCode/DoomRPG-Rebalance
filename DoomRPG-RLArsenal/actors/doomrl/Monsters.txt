// Marine
ACTOR RLCorruptedMarine1RPG : RLCorruptedMarine1 Replaces RLCorruptedMarine1
{
	States
	{
	Spawn:
		XMAR A 0 NoDelay ACS_NamedExecuteAlways("MonsterInit", 0)
		Goto Super::Spawn
	See:
		XMAR A 0 ACS_NamedExecuteAlways("MonsterInit", 0)
		Goto Super::See

}
