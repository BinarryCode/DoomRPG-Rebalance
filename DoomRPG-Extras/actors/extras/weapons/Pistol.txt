Actor Pistol1 : Pistol replaces Pistol
{
	Decal Bulletchip
	
	Weapon.SelectionOrder 190
	Weapon.AmmoGive 10
	Weapon.UpSound "weapons/pistdraw"
	Inventory.pickupsound "weapons/pistol/pickup"
	Inventory.icon "PISTA0"
	
	AttackSound "weapons/pistol/fire"
	
	States
	{
	Ready:
		PISG A 1 A_WeaponReady
		Loop
	Deselect:
		TNT1 A 0 A_lower
		PISG A 1 A_Lower
		Loop
	Select:
		PISG A 1 A_Raise
		Loop
	Fire:
		TNT1 A 0 A_GunFlash
		PISF A 3 Bright A_FireBullets(2, 0, 1, 5, "BulletPuffed")
		PISG A 1
		PISG B 1
		PISG C 1
		PISG D 1
		PISG E 1
		PISG F 1 A_FireCustomMissile("9mmCasingSpawner", 0, 0, 8, -4)
		PISG G 2
		PISG A 2 
		TNT1 A 0 A_ReFire
		PISG A 5 A_FireCustomMissile("SmokeSpawner", 0, 0, 0, -2)
		Goto Ready
	AltFire:
		TNT1 A 0 A_GunFlash
		PISF A 2 Bright A_FireBullets(5.6, 0, 1, 5, "BulletPuffed") //exact equivalent of A_FirePistol
		PISG A 1
		PISG B 1
		PISG C 1
		PISG D 1
		PISG E 1
		PISG F 1 A_FireCustomMissile("9mmCasingSpawner", 0, 0, 8, -4)
		PISG G 0
		PISG A 0 
		TNT1 A 0 A_ReFire
		PISG A 2 A_FireCustomMissile("SmokeSpawner", 0, 0, 0, -2)
		Goto Ready
	Flash:
		TNT1 A 3 A_Light2 
		TNT1 A 1 A_light1
		goto LightDone
	}
}
